### 2. Quotation Engine

The core quotation calculation engine.

```python
# motor_cotizacion_panelin.py

class MotorCotizacionPanelin:
    """Core quotation calculation engine"""
    
    def calcular_cotizacion(
        self,
        producto: str,
        espesor: str,
        largo: float,
        ancho: float,
        tipo_fijacion: str,
        luz: float,
        alero_1: float = 0,
        alero_2: float = 0
    ) -> Dict:
        """Calculate complete quotation with validation"""
        
        # 1. Validate autoportancia (load-bearing capacity)
        validacion = self._validar_autoportancia(producto, espesor, luz)
        
        # 2. Calculate area and materials
        area = largo * ancho
        materiales = self._calcular_materiales(producto, espesor, area, tipo_fijacion)
        
        # 3. Get prices from knowledge base
        precios = self._obtener_precios(producto, espesor)
        
        # 4. Calculate costs with IVA (22%)
        costos = self._calcular_costos(materiales, precios)
        
        return {
            'producto': producto,
            'espesor': espesor,
            'dimensiones': {'largo': largo, 'ancho': ancho, 'area': area},
            'validacion': validacion,
            'materiales': materiales,
            'costos': costos
        }
```

### 3. Source of Truth Validator

Validates data consistency across knowledge base levels.

```python
# panelin_improvements/source_of_truth_validator.py

class SourceOfTruthValidator:
    """Validates source of truth hierarchy"""
    
    HIERARCHY = {
        1: "BMC_Base_Conocimiento_GPT-2.json",  # Master
        2: "BMC_Base_Unificada_v4.json",         # Validation
        3: "panelin_truth_bmcuruguay_web_only_v2.json",  # Dynamic
        4: ["Aleros.rtf", "*.csv", "*.md"]       # Support
    }
    
    def validate_price(self, product: str, price: float) -> ValidationResult:
        """Validate price against source of truth"""
        master_price = self._get_level1_price(product)
        
        if abs(price - master_price) > 0.01:
            return ValidationResult(
                valid=False,
                message=f"Price mismatch: {price} vs {master_price}",
                source="Level 1 Master"
            )
        
        return ValidationResult(valid=True)
```
